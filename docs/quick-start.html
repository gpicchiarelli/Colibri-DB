Integrazione Swift

Usa Colibr√¨DB nelle tue applicazioni Swift

Esempio Base

main.swift swift 
import ColibriCore

@main
struct ColibriExample {
    static func main() async throws {
        // Configura il database
        let config = DatabaseConfig(
            dataDirectory: "./data",
            bufferPoolSize: 1024 * 1024 * 1024, // 1GB
            walEnabled: true
        )
        
        // Crea e avvia il database
        let db = try Colibr√¨DB(config: config)
        try await db.start()
        
        // Inizia una transazione
        let txID = try await db.beginTransaction()
        
        // Crea una tabella
        let schema = TableSchema(columns: [
            Column(name: "id", type: .integer, primaryKey: true),
            Column(name: "name", type: .string, nullable: false),
            Column(name: "email", type: .string, nullable: true)
        ])
        try await db.createTable(name: "users", schema: schema, txID: txID)
        
        // Inserisci dati
        let row: Row = [
            "id": .int(1),
            "name": .string("Alice"),
            "email": .string("alice@example.com")
        ]
        try await db.insert(table: "users", row: row, txID: txID)
        
        // Conferma la transazione
        try await db.commit(txID)
        
        // Esegui una query
        let result = try await db.execute("""
            SELECT name, email FROM users WHERE id = 1
        """, txID: try await db.beginTransaction())
        
        for row in result.rows {
            print("Name: \(row["name"]), Email: \(row["email"])")
        }
        
        // Arresta il database
        try await db.stop()
    }
}
